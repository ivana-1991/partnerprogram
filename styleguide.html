Dismiss animacija (JS)
closeBtn.addEventListener('click', () => {
  const al = closeBtn.closest('.alert');
  al.style.transition = 'opacity .2s, max-height .25s .05s, margin .25s .05s, padding .25s .05s';
  al.style.overflow   = 'hidden';
  al.style.maxHeight  = al.offsetHeight + 'px';
  al.style.opacity    = '0';
  requestAnimationFrame(() => {
    al.style.maxHeight = '0';
    al.style.padding   = '0 20px';
    al.style.margin    = '0';
  });
  setTimeout(() => al.remove(), 320);
});
